    const dilemmaOutput = document.getElementById('dilemma-output');

    const loadingIndicator = document.getElementById('loading-indicator');

 

    analyzeDilemmaBtn.addEventListener('click', async () => {

        const dilemma = dilemmaInput.value.trim();

        const selectedTheory = theorySelect.value;

 

        if (!dilemma) {

            dilemmaOutput.innerHTML = '<p class="text-red-500">Please enter an ethical dilemma.</p>';

            return;

        }

 

        loadingIndicator.classList.remove('hidden');

        dilemmaOutput.innerHTML = '';

 

        try {

            let chatHistory = [];

            const prompt = `Analyze the following ethical dilemma from the perspective of ${selectedTheory}. Explain how this theory would approach the situation and what action it might suggest.

            Dilemma: ${dilemma}`;

           

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

 

            const response = await fetch(apiUrl, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify(payload)

            });

            const result = await response.json();

 

            if (result.candidates && result.candidates.length > 0 &&

                result.candidates[0].content && result.candidates[0].content.parts &&

                result.candidates[0].content.parts.length > 0) {

                const text = result.candidates[0].content.parts[0].text;

                dilemmaOutput.innerHTML = `<p>${text}</p>`;

            } else {

                dilemmaOutput.innerHTML = '<p class="text-red-500">Could not get a response. Please try again.</p>';

            }

        } catch (error) {

            console.error('Error calling Gemini API:', error);

            dilemmaOutput.innerHTML = '<p class="text-red-500">An error occurred while analyzing the dilemma. Please check your network connection or try again later.</p>';

        } finally {

            loadingIndicator.classList.add('hidden');

        }

    });
